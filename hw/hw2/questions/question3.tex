        \Procname{$\proc{FindIntersection}(A_n, B_n, n)$}
        \li $I =$ empty list
        \li $a = 0$, $b = 0$, $most\_recent = -1$
        \li \While $a < n$ and $b<n$
            \Do
                \li \If $A[a]$ is $B[b]$
                \Then
                    \li \If $A[a]$ is not $most\_recent$
                    \Then
                        \li $I.insert(A[a])$
                        \li $most\_recent = A[a]$
                    \End
                    \li $a = a + 1$
                    \li $b = b + 1$
                \li \ElseIf $A[a] < B[b]$
                \Then
                    \li $a = a + 1$
                \li \ElseIf $A[a] > B[b]$
                \Then
                    \li $b = b + 1$
                \End 
            \End
    \end{codebox}
