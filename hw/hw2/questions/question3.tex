    \begin{spacing}{1.2}
    \subsection{(a)} 
        Assume $S_n$ represents sequence $S$ with size of $n$ elements:
        \begin{codebox}
            \Procname{$\proc{FindIntersection}(A_n, B_n, n)$}
            \li $I =$ empty list
            \li $a = 0$, $b = 0$, $most\_recent = -1$
            \li \While $a < n$ and $b<n$
                \Do
                    \li \If $A[a]$ is $B[b]$
                    \Then
                        \li \If $A[a]$ is not $most\_recent$
                        \Then
                            \li $I.insert(A[a])$
                            \li $most\_recent = A[a]$
                        \End
                        \li $a = a + 1$
                        \li $b = b + 1$
                    \li \ElseIf $A[a] < B[b]$
                    \Then
                        \li $a = a + 1$
                    \li \ElseIf $A[a] > B[b]$
                    \Then
                        \li $b = b + 1$
                    \End 
                \End
            \li \If $a$ is $n$
            \Then
                \li \For $b$ to $n-1$
                \Then
                    \li \If $A[n-1]$ is $B[b]$ and $B[b]$ is not $most\_recent$
                    \Then
                        \li $I.insert(B[b])$
                        \li $most\_recent = B[b]$
                    \End
                \End
            \li \ElseIf $b$ is $n$
            \Then
                \li \For $a$ to $n-1$
                \Then
                    \li \If $A[a]$ is $B[n-1]$ and $A[a]$ is not $most\_recent$
                    \Then
                        \li $I.insert(A[a])$
                        \li $most\_recent = A[a]$
                    \End
                \End
            \End
            \li \Return $I$
        \end{codebox}
    \end{spacing}
